### Veri Yapısı (JSON Schema)
```typescript
interface YapiTuruData {
  // SEO Bilgileri
  seo: {
    title: string;              // "Bağ Evi Nasıl Yapılır? | Tarım İmar"
    description: string;        // Meta description
    keywords: string[];         // Anahtar kelimeler
    canonical: string;          // "/bag-evi"
    ogImage?: string;          // Özel OG image
  };
  
  // Landing Page İçeriği
  hero: {
    title: string;              // "Bağ Evi Yapımı ve Hesaplama"
    subtitle: string;           // Kısa açıklama
    icon: string;               // Emoji icon
    backgroundImage?: string;  // Hero background
  };
  
  // Özellikler ve Gereksinimler
  features: {
    title: string;
    description: string;
    icon: string;
  }[];
  
  requirements: {
    legal: {                    // Yasal gereksinimler
      minArea?: number;         // Minimum alan (m² veya dekar)
      maxBuildingArea?: number; // Max inşaat alanı
      restrictions: string[];   // Kısıtlamalar
      genelge: string[];        // Genelge maddeleri
    };
    technical: {                // Teknik gereksinimler
      specifications: string[]; // Teknik özellikler
      capacity?: any;           // Kapasite bilgileri
    };
  };
  
  // Hesaplama Bölümü
  calculator: {
    title: string;
    description: string;
    ctaText: string;            // "Bağ Evi Hesapla"
    ctaLink: string;            // "/hesaplama/bag-evi"
  };
  
  // SSS (FAQ)
  faq: {
    question: string;
    answer: string;
  }[];
  
  // Structured Data
  structuredData: any;          // Schema.org JSON-LD
  
  // Veri Kaynakları
  sources: {
    genelge: boolean;           // Genelge'den alındı mı?
    pythonModule: boolean;      // Python modülünden alındı mı?
    webScraping: boolean;       // Web scraping yapıldı mı?
    scrapedUrls?: string[];    // Scrape edilen URL'ler
    lastUpdated: string;        // Son güncelleme tarihi
  };
}
```

export async function getStaticPaths() {
  const yapiTurleri = await getAllYapiTurleri();
  return {
    paths: yapiTurleri.map(yapi => ({
      params: { 'yapi-turu': yapi.slug }
    })),
    fallback: false
  };
}

export async function getStaticProps({ params }) {
  const data = await getYapiTuruData(params['yapi-turu']);
